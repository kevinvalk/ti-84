// Solution to `X^A MOD N` given in T
// Writes to B,I,T
If F≠666
Then
  // Setup Graph screen
  AxesOff
  ClrDraw

  // Get input
  Disp "X^A MOD N"
  Input "X=",X
  Input "A=",A
  Input "N=",N
End
A→B

// Main loop
0→I
1→T
While A≠0
  T*X→T
  remainder(T,N)→T
  A-1→A
End

If F≠666
Then
  // Output
  Text(0,0,X,"^",B," mod ",N,"=")
  Text(6,0,T)
End
